
Run a container
kubectl run nginx --image=nginx

kubectl get pods

kubectl run part-inventory-pod --image=ram1uj/part-inventory-service

kubectl get pods

kubectl get pods -o wide

get inside the nginx container:
kubectl exec -it nginx -- sh

or

kubectl exec -it nginx -- curl 10.1.0.7:8080/api/parts


expose the pod as a service:
kubectl expose pod part-inventory-pod --port=80 --target-port=8080


Delete the service
kubectl delete service part-inventory-service

Create a NodePort service
kubectl expose pod part-inventory-pod --port=80 --target-port=8080 --type=NodePort

kubectl get services

for kind cluster:
kubectl port-forward service/part-inventory-pod 8081:80

delete everything
kubectl delete pods --all
kubectl delete services --all


create a deployment
kubectl create deployment part-inventory-deployment --image=ram1uj/part-inventory-service

kubectl get deployments

kubectl expose deployment part-inventory-deployment --type=NodePort --port=80 --target-port=8080

kubectl get services


kubectl scale deployment part-inventory-deployment --replicas=3

kubectl get pods
